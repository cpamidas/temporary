h1 {
    text-align: center; - формат текста по центру
    text-align: justify; - равномерный по правой и левой стороне
    
    text-decoration: overline; - подчеркнутая линий находистя сверху
    text-decoration: underline; - подчеркнутая линия находится снизу
    text-decoration: none; - убирает подчеркивание ссылок
    text-decoration: line-throgh - зачеркнутый текст;
    text-decoration: blink
}
 
p {
    text-indent: 12px; - отступ первой строки
    word-spacing: 20px; - отступ между словами
    letter-spacing: 20px; - отступ между буквами
    
    line-height: 20px; - высота строки - межстрочный интервал
    text-shadow: 2px 4px 2px #cce      - тень для текста (координаты по х и у | радиус размытия | цвет)
    text-overflow: ellipsis(многоточие) - текст, который не поместился в блок, его настройки, можно запрятать внутри блока;
